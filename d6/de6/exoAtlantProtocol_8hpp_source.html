11inline uint32_t CRC32_get(const uint8_t* buf, const uint size) {
12 uint32_t crc = ~0;
13 for (uint32_t i = 0; i < size; ++i) {
14 crc ^= buf[i] << (i % 24);
15 // crc =~crc;
16 }
17 return ~crc;
18}

132using ver_t = uint8_t;

136struct pkt_desc_t {
137 ver_t ver;
138 type_t type;
139 sp_t sp;
140 addr_t addr_dst;
141 addr_t addr_src;
142 exo_res_t
143 res;
144 uint8_t* data;
145 uint maxsize;
146 uint size;
147 // дополнил для тестов
148 bool operator==(const pkt_desc_t& rhs) const {
149 if (this->ver != rhs.ver) return false;
150 if (this->type != rhs.type) return false;
151 if (this->sp != rhs.sp) return false;
152 if (this->addr_dst != rhs.addr_dst) return false;
153 if (this->addr_src != rhs.addr_src) return false;
154 if (this->res != rhs.res) return false;
155 if (this->size != rhs.size) return false;
156 if (type == PACKET1) {
157 if (*(Packet1*)(data) != *(Packet1*)(rhs.data)) return false;
158 }
159 if (type == PACKET2) {
160 if (*(Packet2*)(data) != *(Packet2*)(rhs.data)) return false;
161 }
162 if (type == PACKET3) {
163 if (*(Packet3*)(data) != *(Packet3*)(rhs.data)) return false;
164 }
165
166 return true;
167 }

210 pkt_desc_t parse(const uint8_t* buf, uint size) {
211 pkt_desc_t result{};
212 CustomSpan<uint8_t> span(buf, size);
213 size_t read = 0;

218 result.type = meta::get_named<FieldName::TYPE_FIELD>(snap);

220 result.addr_dst = meta::get_named<FieldName::DEST_FIELD>(snap);
221 result.addr_src = meta::get_named<FieldName::SOURCE_FIELD>(snap);
222 result.res = meta::get_named<FieldName::ANS_TYPE_FIELD>(snap);
223 auto data_field = meta::get_named<FieldName::DATA_FIELD>(snap);
224 if (std::holds_alternative<Packet1>(data_field)) {
225 auto tmp = std::get<Packet1>(data_field);
226 result.data = (uint8_t*)&trank1;
227 memcpy(result.data, &tmp, sizeof(Packet1));
228 result.size = sizeof(Packet1);
229 }
230 if (std::holds_alternative<Packet2>(data_field)) {
231 auto tmp = std::get<Packet2>(data_field);
