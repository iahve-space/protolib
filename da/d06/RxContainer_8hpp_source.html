261 template <size_t Index>
262 auto fill_fields(CustomSpan<uint8_t>& ptr, size_t& read) -> MatchStatus {
263 auto& field = std::get<Index>(this->m_fields);
264 if (field.get_size() == 0) {
265 return MatchStatus::MATCH;
266 }
267 size_t byte_to_read =
268 std::min(ptr.size(), field.get_size() - field.m_read_count);
269 if (FieldTraits<decltype(field)>::CONST_VALUE != nullptr) {
270 if constexpr (has_flag(std::remove_reference_t<decltype(field)>::FLAGS,
271 FieldFlags::REVERSE)) {
272 for (int i = 0; i < static_cast<int>(byte_to_read); i++) {
273 if (ptr[i] != field.CONST_VALUE[field.get_size() - 1 -
274 field.m_read_count - i]) {
275 ++read;
276 if (this->is_debug()) {
277 }
278 return MatchStatus::NOT_MATCH;
279 }
280 }
281 } else {
282 if (std::memcmp(ptr.data(),
283 (uint8_t*)field.CONST_VALUE + field.m_read_count,
284 byte_to_read) != 0) {
285 ++read;
286 if (this->is_debug()) {
287 }
288 return MatchStatus::NOT_MATCH;
289 }
290 }
291 }
292 read += byte_to_read;
293 if constexpr (has_flag(std::remove_reference_t<decltype(field)>::FLAGS,
294 FieldFlags::REVERSE)) {
295 for (int i = 0; i < static_cast<int>(byte_to_read); i++) {
296 uint8_t* data = (field.BASE + field.m_offset + field.get_size() - 1) -
297 field.m_read_count - i;
298 *data = ptr.data()[i];
299 }
300 } else {
301 std::memcpy(field.BASE + field.m_offset + field.m_read_count, ptr.data(),
302 byte_to_read);
303 }
304 field.m_read_count += byte_to_read;
305
306 if (field.m_read_count < field.m_size) {
307 return MatchStatus::PROCESSING;
308 }
309 if (field.m_matcher) {
310 return field.m_matcher(static_cast<void*>(this));
311 }
312 field.m_read_count = 0;
313 return MatchStatus::MATCH;
314 }

423 static auto check_crc(void* obj) -> MatchStatus {
424 auto& container = *static_cast<RxContainer<Fields, TCrc>*>(obj);
425 auto crc_in_field =
426 *container.template get<FieldName::CRC_FIELD>().get_ptr();
427 using crc_type = decltype(crc_in_field);
428 uint32_t crc = 0;
429 container.m_crc.reset();
430
431 container.for_each_type([&](auto& field) {
432 using field_type = std::remove_reference_t<decltype(field)>;
433 if constexpr (has_flag(field_type::FLAGS, FieldFlags::IS_IN_CRC)) {
434 auto* data = field.get_ptr();
435 size_t size = field.get_size();
436 crc = container.m_crc.append(crc, {(uint8_t*)data, size});
437 }
438 });
439
440 bool result = crc_in_field == static_cast<decltype(crc_in_field)>(crc);
441 if (container.is_debug() && not result) {
442 const auto FLAGS = std::cout.flags(); // сохранить текущие флаги
443 const auto FILL = std::cout.fill(); // и текущий fill-символ
444
445 auto to_u = [](const auto VAL) -> uint64_t {
446 return static_cast<uint64_t>(VAL);
447 };
448 const crc_type CRC = -1;
449 const auto CRC_EXP = to_u(crc & CRC);
