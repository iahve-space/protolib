75 // Permanent RX-callback that snapshots only the last type code.
76 // NOTE: We intentionally do NOT materialize a variant for DATA_FIELD here,

79 m_rx_delegate = m_rx.add_receive_callback(m_rx_callback);
80 m_deque_thread = std::thread([this] {
81 std::unique_lock<std::mutex> lock(m_queue_mutex);
82 while (true) {
83 m_queue_cv.wait(lock,
84 [this] { return !m_running || !m_rx_queue.empty(); });
85 if (!m_running) {
86 break;
87 }
88 auto val = std::move(m_rx_queue.front());
89 // вызываем без мьютекса

91 if (m_user_callback) {

93 m_rx_queue.pop_front();
94 }
95 lock.lock();
96 }
97 });
