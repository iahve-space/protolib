187 if constexpr (field_info_has_name<FieldType::NAME, InfoTuple>()) {
188 auto& data_info = get_field_info_by_name<FieldType::NAME>(info_tuple);
189 field.set((void*)data_info.m_data);

191 if (field.m_matcher != nullptr) {
192 field.m_matcher(this);

194 field.apply_const();

196 this->m_offsets[field.BASE] += field.m_size;
