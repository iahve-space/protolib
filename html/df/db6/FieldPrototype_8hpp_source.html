182 [[nodiscard]] auto get_copy() const -> CopyType {
183 if constexpr (IS_STD_VECTOR_V<CopyType>) {
184 // T — указатель ⇒ возвращаем вектор элементов
185 const std::size_t FIELD_SIZE = get_size(); // сколько байт в поле
186 const std::size_t COUNT = FIELD_SIZE / sizeof(Elem); // сколько элементов
187 CopyType val; // это std::vector<Elem>
188 val.resize(COUNT);
189 if (COUNT != 0U) {
190 // предполагаем, что данные плоские и совместимы по представлению
191 std::memcpy(val.data(), get_ptr(), COUNT * sizeof(Elem));
192 }
193 return val;
194 } else {
195 // T — не указатель ⇒ вернуть T по значению
196 CopyType value{};
197 if constexpr (std::is_trivially_copyable_v<CopyType>) {
198 // безопасно побайтно
199 std::memcpy(&value, get_ptr(), sizeof(CopyType));
200 } else {
201 // Лучше явно сконструировать из байтов, если есть такой конструктор,
202 // или написать парсер; memcpy для нетривиальных типов — UB.

301 constexpr auto set_size(size_t size_to_set = K_ANY_SIZE) -> bool {
302 if constexpr (CONST_VALUE != nullptr) {
303 static_assert(CONST_VALUE == nullptr,
304 "You can't set size for fields with const_value");
305 }
306 if constexpr (SIZE == K_ANY_SIZE) {
307 if constexpr (!std::is_same_v<std::remove_pointer_t<DATA>,

309 constexpr size_t K_STATIC_SIZE =
310 std::is_pointer_v<DATA> ? 0 : sizeof(DATA);
311 if (size_to_set != K_ANY_SIZE) {
312 if (size_to_set <= MAX_SIZE) {
313 this->m_size = size_to_set;
314 } else {
315 return false;
316 }
317 } else if constexpr (K_STATIC_SIZE > 0) {
318 this->m_size = K_STATIC_SIZE;
319 }
320 } else {
321 this->m_size = 0;
322 }
323 }
324 return true;
