15inline auto parse_uint_sv(std::string_view string_view) -> uint64_t {
16 // trim spaces
17 const auto LEFT = string_view.find_first_not_of(" \t\n\r");
18 const auto RIGHT = string_view.find_last_not_of(" \t\n\r");
19 if (LEFT == std::string_view::npos) {
20 return 0;
21 }
22 string_view = string_view.substr(LEFT, RIGHT - LEFT + 1);
23
24 // remove common separators for convenience
25 std::string str;
26 str.reserve(string_view.size());
27 for (const char VAL : string_view) {
28 if (VAL == ' ' || VAL == '\t' || VAL == '\n' || VAL == '_') {
29 continue;
30 }
31 str.push_back(VAL);
32 }
33 constexpr static uint8_t DECIMAL_BASE = 10;
34 constexpr static uint8_t HEX_BASE = 16;
35 int base = DECIMAL_BASE;
36 if (str.size() > 2 && str[0] == '0' && (str[1] == 'x' || str[1] == 'X')) {
37 base = HEX_BASE;
38 } else {
39 for (const char VAL : str) {
40 if ((VAL >= 'a' && VAL <= 'f') || (VAL >= 'A' && VAL <= 'F')) {
41 base = HEX_BASE;
42 break;
43 }
44 }
45 }
46
47 try {
48 size_t idx = 0;
49 const unsigned long long VAL = std::stoull(str, &idx, base);
50 (void)idx;
51 return static_cast<uint64_t>(VAL);
52 } catch (...) {
53 return 0;
54 }
55}

57inline void parse_hex_bytes_fill(const std::string_view STR_VIEW, uint8_t* out,
58 const size_t BYTES) {
59 constexpr static uint8_t DECIMAL_BASE = 10;
60 std::string hex;
61 hex.reserve(STR_VIEW.size());
62 for (size_t i = 0; i < STR_VIEW.size(); ++i) {
63 const char VAL = STR_VIEW[i];
64 if (VAL == '0' && i + 1 < STR_VIEW.size() &&
65 (STR_VIEW[i + 1] == 'x' || STR_VIEW[i + 1] == 'X')) {
66 ++i; // пропустить 'x'/'X'
67 continue; // и не добавлять '0' от префикса
68 }
69 if (std::isxdigit(static_cast<unsigned char>(VAL)) != 0) {
70 hex.push_back(VAL);
71 }
72 }
73 // дальше как было:
74 if ((hex.size() % 2) != 0U) {
75 hex.insert(hex.begin(), '0');
76 }
77 std::fill_n(out, BYTES, 0);
78 const size_t PAIRS = std::min(BYTES, hex.size() / 2);
79 for (size_t i = 0; i < PAIRS; ++i) {
80 const auto HIGH = hex[2 * i];
81 const auto LOW = hex[(2 * i) + 1];
82 auto val = [](const char VAL) {
83 if (VAL >= '0' && VAL <= '9') {
84 return VAL - '0';
85 }
86 if (VAL >= 'a' && VAL <= 'f') {
87 return DECIMAL_BASE + (VAL - 'a');
88 }
89 if (VAL >= 'A' && VAL <= 'F') {
90 return DECIMAL_BASE + (VAL - 'A');
91 }
92 return 0;
93 };
94 out[i] = static_cast<uint8_t>((val(HIGH) << 4) | val(LOW));
95 }
96}
